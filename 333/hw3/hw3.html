# based on pseudocode from: # http://en.wikipedia.org/wiki/Insertion_sort insertion_sort = (videos) -> for j in [1..videos.length-1] key = videos[j] i = j - 1 while i >= 0 and compare(videos[i], key) < 0 videos[i+1] = videos[i] i-- videos[i+1] = key return videos # similar to compareTo, but based on video rating compare = (video1, video2) -> views1 = video1.gd$rating.average views2 = video2.gd$rating.average if views1 < views2 return -1 else if views1 > views2 return 1 else return 0 print_info = (video) -> p "-----" p "#{video.title.$t}" p "Rating: #{video.gd$rating.average}" p "Views: #{video.yt$statistics.viewCount}" on_get_json = (data) -> sorted_videos = insertion_sort data.feed.entry print_info video for video in sorted_videos p "Loading most viewed YouTube videos via AJAX..." $.getJSON "https://gdata.youtube.com/feeds/api/standardfeeds/on_the_web?v=2&alt=json", on_get_json
